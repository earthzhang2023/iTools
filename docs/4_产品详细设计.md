### 2.1 主题管理模块

#### 2.1.1 功能描述

#### 2.1.2 数据结构

```typescript
}

  common: {
    primaryColor: string
    successColor: string
    warningColor: string
    errorColor: string
  }
}
```

#### 2.1.3 接口设计
```typescript

interface UseTheme {
  // 切换主题模式
  toggleTheme: () => void
  
  // 设置主题模式
  setThemeMode: (mode: 'light' | 'dark' | 'auto') => void
  
  // 获取当前主题
  getTheme: () => ThemeState
```

#### 2.1.4 实现流程
```
1. 应用启动
   ↓
2. 从 localStorage 读取主题配置
   ↓
3. 如果没有配置，检测系统主题
   ↓
4. 应用主题配置到 Naive UI
   ↓
5. 更新 CSS 变量
   ↓
6. 监听系统主题变化（如果设置为 auto）
```

#### 2.1.5 关键代码
```typescript
// stores/theme.ts
import { defineStore } from 'pinia'
import { darkTheme } from 'naive-ui'

export const useThemeStore = defineStore('theme', {
  state: (): ThemeState => ({
    mode: 'auto',
    isDark: false,
    overrides: {
      common: {
        primaryColor: '#3b82f6',
        successColor: '#10b981',
        warningColor: '#f59e0b',
        errorColor: '#ef4444'
      }
    }
  }),
  
  actions: {
    initTheme() {
      const saved = localStorage.getItem('theme-mode')
      this.mode = (saved as 'light' | 'dark' | 'auto') || 'auto'
      this.applyTheme()
      this.watchSystemTheme()
    },
    
    applyTheme() {
      if (this.mode === 'auto') {
        this.isDark = window.matchMedia('(prefers-color-scheme: dark)').matches
      } else {
        this.isDark = this.mode === 'dark'
      }
      localStorage.setItem('theme-mode', this.mode)
    },
    
    watchSystemTheme() {
      const mediaQuery = window.matchMedia('(prefers-color-scheme: dark)')
      mediaQuery.addEventListener('change', () => {
        if (this.mode === 'auto') {
          this.applyTheme()
        }
      })
    }
  }
})
```

```typescript
// tools/base64/index.ts
interface Base64EncodeOptions {
  urlSafe?: boolean  // 是否使用 URL 安全字符
}

export function base64Encode(
  input: string, 
  options: Base64EncodeOptions = {}
): string {
  try {
    const encoded = btoa(encodeURIComponent(input))
    return options.urlSafe 
      ? encoded.replace(/\+/g, '-').replace(/\//g, '_')
      : encoded
  } catch (error) {
    throw new Error('Base64 编码失败：输入包含无效字符')
  }
}
```

#### 3.1.2 解码函数
```typescript
export function base64Decode(input: string): string {
  try {
    // 处理 URL 安全字符
    const normalized = input.replace(/-/g, '+').replace(/_/g, '/')
    return decodeURIComponent(atob(normalized))
  } catch (error) {
    throw new Error('Base64 解码失败：输入不是有效的 Base64 字符串')
  }
}
```

### 3.2 JSON 工具函数

#### 3.2.1 格式化函数
```typescript
// tools/json/index.ts
interface JsonFormatOptions {
  indent?: number    // 缩进空格数，默认 2
  sortKeys?: boolean // 是否排序键，默认 false
}

export function jsonFormat(
  input: string, 
  options: JsonFormatOptions = {}
): string {
  const { indent = 2, sortKeys = false } = options
  
  try {
    const parsed = JSON.parse(input)
    const replacer = sortKeys 
      ? (key: string, value: any) => 
          value instanceof Object && !(value instanceof Array) 
            ? Object.keys(value).sort().reduce((sorted, key) => {
                sorted[key] = value[key]
                return sorted
              }, {} as any)
            : value
      : undefined
    
    return JSON.stringify(parsed, replacer, indent)
  } catch (error) {
    throw new Error(`JSON 格式化失败：${error.message}`)
  }
}
```

#### 3.2.2 验证函数
```typescript
interface JsonValidateResult {
  valid: boolean
  error?: {
    message: string
    line?: number
    column?: number
    position?: number
  }
}

export function jsonValidate(input: string): JsonValidateResult {
  try {
    JSON.parse(input)
    return { valid: true }
  } catch (error) {
    // 尝试解析错误位置
    const match = error.message.match(/position (\d+)/i)
    const position = match ? parseInt(match[1]) : undefined
    
    // 计算行列号
    let line, column
    if (position !== undefined) {
      const lines = input.substring(0, position).split('\n')
      line = lines.length
      column = lines[lines.length - 1].length + 1
    }
    
    return {
      valid: false,
      error: {
        message: error.message,
        line,
        column,
        position
      }
    }
  }
}
```

#### 3.2.3 JSON 转 CSV
```typescript
export function jsonToCsv(input: string): string {
  try {
    const data = JSON.parse(input)
    
    if (!Array.isArray(data) || data.length === 0) {
      throw new Error('JSON 必须是对象数组')
    }
    
    // 获取所有键
    const headers = Array.from(
      new Set(data.flatMap(obj => Object.keys(obj)))
    )
    
    // 生成 CSV
    const csvRows = [
      headers.join(','), // 表头
      ...data.map(row => 
        headers.map(field => {
          const value = row[field] ?? ''
          // 处理包含逗号或引号的值
          if (typeof value === 'string' && (value.includes(',') || value.includes('"'))) {
            return `"${value.replace(/"/g, '""')}"`
          }
          return value
        }).join(',')
      )
    ]
    
    return csvRows.join('\n')
  } catch (error) {
    throw new Error(`JSON 转 CSV 失败：${error.message}`)
  }
}
```

#### 3.2.4 JSON 转 TypeScript 接口
```typescript
export function jsonToTypeScript(input: string, interfaceName = 'Root'): string {
  try {
    const data = JSON.parse(input)
    
    function getType(value: any, name: string): string {
      if (value === null) return 'null'
      if (Array.isArray(value)) {
        if (value.length === 0) return 'any[]'
        const itemType = getType(value[0], `${name}Item`)
        return `${itemType}[]`
      }
      if (typeof value === 'object') {
        const properties = Object.entries(value)
          .map(([key, val]) => {
            const type = getType(val, `${name}${key.charAt(0).toUpperCase() + key.slice(1)}`)
            return `  ${key}: ${type}`
          })
          .join(';\n')
        return `{\n${properties}\n}`
      }
      return typeof value
    }
    
    const type = getType(data, interfaceName)
    return `interface ${interfaceName} ${type}`
  } catch (error) {
    throw new Error(`JSON 转 TypeScript 失败：${error.message}`)
  }
}
```

### 3.3 文本处理函数

#### 3.3.1 大小写转换
```typescript
// tools/text/index.ts
type CaseType = 'upper' | 'lower' | 'title' | 'sentence' | 'camel' | 'pascal' | 'snake' | 'kebab'

export function caseConvert(input: string, type: CaseType): string {
  switch (type) {
    case 'upper':
      return input.toUpperCase()
    
    case 'lower':
      return input.toLowerCase()
    
    case 'title':
      return input.replace(/\w\S*/g, (txt) =>
        txt.charAt(0).toUpperCase() + txt.substr(1).toLowerCase()
      )
    
    case 'sentence':
      return input.replace(/(^\s*\w|[\.\!\?]\s*\w)/g, (c) => c.toUpperCase())
    
    case 'camel':
      return input.replace(/(?:^\w|[A-Z]|\b\w)/g, (word, index) =>
        index === 0 ? word.toLowerCase() : word.toUpperCase()
      ).replace(/\s+/g, '')
    
    case 'pascal':
      return input.replace(/(?:^\w|[A-Z]|\b\w)/g, (word) =>
        word.toUpperCase()
      ).replace(/\s+/g, '')
    
    case 'snake':
      return input.replace(/\W+/g, '_').toLowerCase()
    
    case 'kebab':
      return input.replace(/\W+/g, '-').toLowerCase()
    
    default:
      return input
  }
}
```

```typescript
interface TextStats {
  length: number          // 总字符数
  charsNoSpace: number    // 字符数（不含空格）
  words: number           // 单词数
  lines: number           // 行数
  chinese: number         // 中文字符数
  english: number         // 英文字母数
  numbers: number         // 数字数
  spaces: number          // 空格数
  special: number         // 特殊字符数
}

export function textStats(input: string): TextStats {
  return {
    length: input.length,
    charsNoSpace: input.replace(/\s/g, '').length,
    words: input.trim() === '' ? 0 : input.trim().split(/\s+/).length,
    lines: input.trim() === '' ? 0 : input.split(/\n/).length,
    chinese: (input.match(/[\u4e00-\u9fa5]/g) || []).length,
    english: (input.match(/[a-zA-Z]/g) || []).length,
    numbers: (input.match(/[0-9]/g) || []).length,
    spaces: (input.match(/\s/g) || []).length,
    special: (input.match(/[^\w\s\u4e00-\u9fa5]/g) || []).length
  }
}
```

### 3.4 时间处理函数

#### 3.4.1 时间戳转换
```typescript
// tools/time/index.ts
interface TimestampOptions {
  unit?: 'seconds' | 'milliseconds'  // 时间戳单位
  format?: string                     // 输出格式
}

export function timestampToDateTime(
  timestamp: number | string, 
  options: TimestampOptions = {}
): string {
  const { unit = 'milliseconds', format = 'YYYY-MM-DD HH:mm:ss' } = options
  
  // 如果是秒，转换为毫秒
  let ts = typeof timestamp === 'string' ? parseInt(timestamp) : timestamp
  if (unit === 'seconds') {
    ts *= 1000
  }
  
  const date = new Date(ts)
  
  if (isNaN(date.getTime())) {
    throw new Error('无效的时间戳')
  }
  
  // 格式化日期
  const year = date.getFullYear()
  const month = String(date.getMonth() + 1).padStart(2, '0')
  const day = String(date.getDate()).padStart(2, '0')
  const hours = String(date.getHours()).padStart(2, '0')
  const minutes = String(date.getMinutes()).padStart(2, '0')
  const seconds = String(date.getSeconds()).padStart(2, '0')
  
  return format
    .replace('YYYY', String(year))
    .replace('MM', month)
    .replace('DD', day)
    .replace('HH', hours)
    .replace('mm', minutes)
    .replace('ss', seconds)
}

export function dateTimeToTimestamp(dateTime: string): number {
  const timestamp = new Date(dateTime).getTime()
  
  if (isNaN(timestamp)) {
    throw new Error('无效的日期时间格式')
  }
  
  return timestamp
}
```

#### 3.5.3 JWT 解析
```typescript
interface JWTPayload {
  header: any
  payload: any
  signature: string
  valid: boolean
  error?: string
}

export function parseJWT(token: string): JWTPayload {
  try {
    const parts = token.split('.')
    
    if (parts.length !== 3) {
      throw new Error('无效的 JWT 格式')
    }
    
    const [header, payload, signature] = parts
    
    return {
      header: JSON.parse(atob(header.replace(/_/g, '/').replace(/-/g, '+'))),
      payload: JSON.parse(atob(payload.replace(/_/g, '/').replace(/-/g, '+'))),
      signature,
      valid: true
    }
  } catch (error) {
    return {
      header: null,
      payload: null,
      signature: '',
      valid: false,
      error: error.message
    }
  }
}
```

```vue
import { useRouter } from 'vue-router'
import { Search } from '@vicons/ionicons5'
import { tools, toolCategories } from '@/config/tools'
import ToolCard from '@/components/tools/ToolCard.vue'

const router = useRouter()
const searchQuery = ref('')

const filteredCategories = computed(() => {
  if (!searchQuery.value) {
    return toolCategories

  const query = searchQuery.value.toLowerCase()
  const matchedToolIds = tools
    .filter(tool => 
      tool.name.toLowerCase().includes(query) ||

      tool.description.toLowerCase().includes(query) ||
      tool.keywords.some(keyword => keyword.toLowerCase().includes(query))
```

```typescript
import { defineStore } from 'pinia'

export const useThemeStore = defineStore('theme', {
  state: () => ({
    mode: 'auto' as 'light' | 'dark' | 'auto',
    isDark: false
  }),
  
  getters: {
    themeConfig: (state) => ({
      theme: state.isDark ? darkTheme : undefined,
      themeOverrides: {
        common: {
          primaryColor: '#3b82f6',
          successColor: '#10b981',
          warningColor: '#f59e0b',
          errorColor: '#ef4444'
        }
      }
    })
  },
  
  actions: {
    init() {
      const saved = localStorage.getItem('theme-mode')
      this.mode = (saved as any) || 'auto'
      this.apply()
      this.watchSystem()
    },
    
    toggle() {
      this.mode = this.mode === 'dark' ? 'light' : 'dark'
      this.apply()
    },
    
    apply() {
      if (this.mode === 'auto') {
        this.isDark = window.matchMedia('(prefers-color-scheme: dark)').matches
      } else {
        this.isDark = this.mode === 'dark'
      }
      localStorage.setItem('theme-mode', this.mode)
    },
    
    watchSystem() {
      window.matchMedia('(prefers-color-scheme: dark)').addEventListener('change', (e) => {
        if (this.mode === 'auto') {
          this.isDark = e.matches
        }
      })
    }
  }
})
```

```typescript
// stores/tools.ts
import { defineStore } from 'pinia'
import { tools, type Tool } from '@/config/tools'

export const useToolsStore = defineStore('tools', {
  state: () => ({
    recentTools: [] as string[], // 最近使用的工具 ID
    favoriteTools: [] as string[] // 收藏的工具 ID
  }),
  
  getters: {
    allTools: () => tools,
    
    getToolById: (state) => (id: string) => {
      return tools.find(tool => tool.id === id)
    },
    
    recentToolsList: (state) => {
      return state.recentTools
        .map(id => tools.find(t => t.id === id))
        .filter(Boolean) as Tool[]
    },
    
    favoriteToolsList: (state) => {
      return state.favoriteTools
        .map(id => tools.find(t => t.id === id))
        .filter(Boolean) as Tool[]
    }
  },
  
  actions: {
    addRecentTool(toolId: string) {
      this.recentTools = [
        toolId,
        ...this.recentTools.filter(id => id !== toolId)
      ].slice(0, 10)
      this.saveToStorage()
    },
    
    toggleFavorite(toolId: string) {
      const index = this.favoriteTools.indexOf(toolId)
      if (index > -1) {
        this.favoriteTools.splice(index, 1)
      } else {
        this.favoriteTools.push(toolId)
      }
      this.saveToStorage()
    },
    
    saveToStorage() {
      localStorage.setItem('recent-tools', JSON.stringify(this.recentTools))
      localStorage.setItem('favorite-tools', JSON.stringify(this.favoriteTools))
    },
    
    loadFromStorage() {
      const recent = localStorage.getItem('recent-tools')
      const favorite = localStorage.getItem('favorite-tools')
      
      if (recent) {
        this.recentTools = JSON.parse(recent)
      }
      if (favorite) {
        this.favoriteTools = JSON.parse(favorite)
      }
    }
  }
})
```
